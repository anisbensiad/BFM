# Example dual protocol transaction commands
# Initialize system
WAIT 100

# AHB transactions
AHB WRITE 0x1000_0000 0xDEADBEEF SINGLE WORD
WAIT 10
AHB READ 0x1000_0000 SINGLE WORD

# AXI transactions
AXI WRITE 0x2000_0000 0xCAFEBABEDEADBEEF INCR 0
WAIT 10
AXI READ 0x2000_0000 INCR 0

# Burst transactions
AHB WRITE 0x3000_0000 0x11111111 INCR4 WORD
AXI BURST_WRITE 0x4000
